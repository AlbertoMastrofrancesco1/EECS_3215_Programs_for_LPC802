#include "LPC802.h"
int main(void) {

	// Turn on the GPIO system.
	SYSCON->SYSAHBCLKCTRL0 |= SYSCON_SYSAHBCLKCTRL0_GPIO0_MASK;

	// loop infinitely
	while(1)

	{
		asm("NOP");
	}
}// end of main
